Node.js 파일시스템 시뮬레이터

이 페이지는 Node.js의 fs 모듈 기능을 브라우저에서 시뮬레이션합니다. 실제 파일시스템에 접근하는 대신 메모리 내 가상 파일시스템을 사용합니다.

예제 코드

// 파일 읽기
fs.readFile('/example.txt', 'utf8', (err, data) => {
  if (err) {
    console.error('Error:', err);
    return;
  }
  console.log(data);
});
// 파일 쓰기
fs.writeFile('/newfile.txt', '안녕하세요, 새 파일입니다!', 'utf8', (err) => {
  if (err) {
    console.error('Error:', err);
    return;
  }
  console.log('파일이 성공적으로 작성되었습니다.');
});
